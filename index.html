// --- CONFIG --- const DB_NAME = "OnyxPortfolioDB"; const STORE_NAME = "Stocks"; const FIELDS = ['SYM','LTP','PNL','PNL%','ROE','PE','PB','MCAP','SECTOR']; const PALETTE = ['#00f2ff','#7000ff','#ff007a','#ffb800','#00ff88','#007aff','#ff4d6d','#f97316','#a3e635']; let MASTER = [], SECTORS = [], COL = {}, filter = null; async function init() { const statusEl = document.getElementById('status'); // 1. Open/Create DB const db = await new Promise(res => { const req = indexedDB.open(DB_NAME, 1); req.onupgradeneeded = e => e.target.result.createObjectStore(STORE_NAME, { keyPath: "SYM" }); req.onsuccess = () => res(req.result); }); // 2. Try to Read Data const tx = db.transaction(STORE_NAME, "readonly"); MASTER = await new Promise(res => { tx.objectStore(STORE_NAME).getAll().onsuccess = e => res(e.target.result); }); // 3. If Empty, Force Hydrate from JSON if (MASTER.length === 0) { statusEl.innerHTML = "Origin Empty. Syncing from fundamentals.json..."; try { const res = await fetch('fundamentals.json'); const json = await res.json(); const stocks = json.stocks || json; const writeTx = db.transaction(STORE_NAME, "readwrite"); const store = writeTx.objectStore(STORE_NAME); Object.keys(stocks).forEach(ticker => { const s = stocks[ticker]; const qty = s.qty || 0; const avg = s.avgBuy || 0; const ltp = s.ltp || 0; const cost = qty * avg; const pnl = qty > 0 ? Math.round((qty * ltp) - cost) : 0; store.put({ SYM: ticker.split('.')[0].toUpperCase(), LTP: ltp, PNL: pnl || '—', 'PNL%': cost > 0 ? ((pnl/cost)*100).toFixed(1)+'%' : '—', ROE: s.roe || 0, PE: s.pe || '—', PB: s.pb || '—', MCAP: s.mcap || 0, SECTOR: s.sector || 'Other' }); }); writeTx.oncomplete = () => { statusEl.innerHTML = "Sync Complete. Reloading..."; location.reload(); }; } catch (e) { statusEl.innerHTML = "CRITICAL: fundamentals.json not found in this folder."; } return; } // 4. If Data exists, Render document.getElementById('k-stat').textContent = "LIVE"; statusEl.style.display = "none"; SECTORS = [...new Set(MASTER.map(s => s.SECTOR))].sort(); SECTORS.forEach((s, i) => COL[s] = PALETTE[i % PALETTE.length]); render(); } // ... Keep your existing render() and tap() functions ... init();
